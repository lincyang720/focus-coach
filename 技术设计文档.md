## 三、游戏详细规格

### 3.1 Number Memory

**Game Logic**:
1. Display a number sequence in the center of the screen (e.g., 3-7-2-9-5)
2. Each number shows for 1 second, with 0.5 second interval
3. After sequence ends, user inputs the remembered numbers
4. Compare user input with correct sequence, calculate score

**Data Structure**:
```typescript
interface NumberMemoryData {
  sequence: number[];        // [3, 7, 2, 9, 5]
  userInput: number[];       // [3, 7, 2, 9]
  correctCount: number;      // 4
  totalLength: number;       // 5
  sequenceSpeed: number;     // 1000ms per number
}
```

**Scoring Algorithm**:
```typescript
function calculateScore(data: NumberMemoryData): number {
  const accuracy = data.correctCount / data.totalLength;
  return Math.round(accuracy * 100); // 0-100
}
```

**Difficulty Adjustment**:
```typescript
function adjustDifficulty(currentLevel: number, accuracy: number): number {
  if (accuracy >= 0.8 && currentLevel < 10) {
    return currentLevel + 1; // level up
  } else if (accuracy < 0.6 && currentLevel > 1) {
    return currentLevel - 1; // level down
  }
  return currentLevel;
}

// Difficulty corresponds to sequence length
function getSequenceLength(level: number): number {
  return 3 + level; // Level 1: 4 digits, Level 10: 13 digits
}
```

### 3.2 Quick Match

**Game Logic**:
1. Display a target in the center (e.g., red circle)
2. Show 4 options below (different color + shape combinations)
3. User taps the matching option
4. 10 seconds per question, 20 questions total

**Data Structure**:
```typescript
interface QuickMatchData {
  target: { color: string; shape: string }; // { color: 'red', shape: 'circle' }
  options: Array<{ color: string; shape: string }>; // 4 options
  selectedIndex: number; // user's selected index
  isCorrect: boolean;
  responseTimeMs: number; // reaction time
}
```

**Scoring Algorithm**:
```typescript
function calculateScore(sessions: QuickMatchData[]): number {
  const correctCount = sessions.filter(s => s.isCorrect).length;
  const avgResponseTime = sessions.reduce((sum, s) => sum + s.responseTimeMs, 0) / sessions.length;
  
  // Accuracy accounts for 70%, speed accounts for 30%
  const accuracyScore = (correctCount / sessions.length) * 70;
  const speedScore = Math.max(0, (2000 - avgResponseTime) / 2000) * 30; // full score within 2 seconds
  
  return Math.round(accuracyScore + speedScore);
}
```

**Difficulty Adjustment**:
```typescript
function getDifficultyParams(level: number) {
  return {
    responseTimeLimit: Math.max(3000, 10000 - level * 700), // Level 1: 10s, Level 10: 3s
    colorCount: Math.min(2 + Math.floor(level / 3), 6), // more colors as difficulty increases
    shapeCount: Math.min(2 + Math.floor(level / 3), 6), // more shapes as difficulty increases
  };
}
```

### 3.3 N-Back Task

**Game Logic**:
1. Display position (3x3 grid) + letter
2. User judges whether current position matches the position N steps back
3. Default N=2 (2-Back)
4. 30 rounds total, 2 seconds per round

**Data Structure**:
```typescript
interface NBackData {
  n: number; // default 2
  positions: Array<{ row: number; col: number }>; // 30 positions
  letters: string[]; // 30 letters
  userResponses: Array<{
    isMatch: boolean; // user's judgment
    actualMatch: boolean; // actual match
    isCorrect: boolean;
    responseTimeMs: number;
  }>;
}
```

**Scoring Algorithm**:
```typescript
function calculateScore(data: NBackData): number {
  const correctResponses = data.userResponses.filter(r => r.isCorrect).length;
  const accuracy = correctResponses / data.userResponses.length;
  return Math.round(accuracy * 100);
}
```

**Difficulty Adjustment**:
```typescript
function adjustNLevel(level: number): number {
  // Level 1-3: 1-Back, Level 4-6: 2-Back, Level 7-10: 3-Back
  if (level <= 3) return 1;
  if (level <= 6) return 2;
  return 3;
}
```

### 3.4 Task Switch

**Game Logic**:
1. Display a number (e.g., 8)
2. Top shows current rule (Rule A: odd/even | Rule B: greater/less than 5)
3. Rule switches every 5 questions
4. 30 questions total

**Data Structure**:
```typescript
interface TaskSwitchData {
  rules: Array<'parity' | 'magnitude'>; // rule for each question
  numbers: number[]; // number for each question
  userResponses: Array<{
    answer: string; // 'odd' | 'even' | 'small' | 'large'
    correctAnswer: string;
    isCorrect: boolean;
    responseTimeMs: number;
    isSwitchTrial: boolean; // whether this is a rule-switch trial
  }>;
}
```

**Scoring Algorithm**:
```typescript
function calculateScore(data: TaskSwitchData): number {
  const correctCount = data.userResponses.filter(r => r.isCorrect).length;
  const accuracy = correctCount / data.userResponses.length;
  return Math.round(accuracy * 100);
}
```

### 3.5 Stroop Test

**Game Logic**:
1. Display a color word (e.g., "RED", but font color is blue)
2. User selects the font color (blue), ignoring the word meaning
3. 30 questions total, 10 seconds per question
4. Color word list: RED, BLUE, GREEN, YELLOW, PURPLE, ORANGE

**Data Structure**:
```typescript
interface StroopData {
  words: Array<{
    text: string; // displayed word (e.g., "RED")
    color: string; // font color (e.g., "blue"), different from text
  }>;
  userResponses: Array<{
    selectedColor: string;
    isCorrect: boolean;
    responseTimeMs: number;
  }>;
}
```

**Scoring Algorithm**:
```typescript
function calculateScore(data: StroopData): number {
  const correctCount = data.userResponses.filter(r => r.isCorrect).length;
  const accuracy = correctCount / data.userResponses.length;
  const avgResponseTime = data.userResponses.reduce((sum, r) => sum + r.responseTimeMs, 0) / data.userResponses.length;
  
  const accuracyScore = accuracy * 70;
  const speedScore = Math.max(0, (3000 - avgResponseTime) / 3000) * 30;
  
  return Math.round(accuracyScore + speedScore);
}
```

### 5.1 Data Preparation

```typescript
async function prepareWeeklyReportData(userId: string, weekStart: Date): Promise<WeeklyReportInput> {
  // Get all game sessions for this week
  const sessions = await db.game_sessions
    .select()
    .where(eq(game_sessions.user_id, userId))
    .where(gte(game_sessions.completed_at, weekStart))
    .where(lt(game_sessions.completed_at, addDays(weekStart, 7)));
  
  // Group stats by game type
  const statsByGame = sessions.reduce((acc, session) => {
    if (!acc[session.game_type]) {
      acc[session.game_type] = {
        count: 0,
        totalScore: 0,
        totalAccuracy: 0,
        improvement: 0,
      };
    }
    acc[session.game_type].count++;
    acc[session.game_type].totalScore += session.score;
    acc[session.game_type].totalAccuracy += session.accuracy;
    
    return acc;
  }, {} as Record<string, GameStats>);
  
  // Calculate improvement trend (compared to last week)
  // ...
  
  return {
    totalSessions: sessions.length,
    totalDurationSeconds: sessions.reduce((sum, s) => sum + s.duration_seconds, 0),
    averageAccuracy: sessions.reduce((sum, s) => sum + s.accuracy, 0) / sessions.length,
    statsByGame,
    bestDayOfWeek: 'Monday', // best performing day
    suggestions: [],
  };
}
```

### 5.3 AI Call Example

```typescript
async function generateWeeklyReport(userId: string, weekStart: Date) {
  const data = await prepareWeeklyReportData(userId, weekStart);
  
  const prompt = WEEKLY_REPORT_PROMPT
    .replace('{totalSessions}', data.totalSessions.toString())
    .replace('{totalDurationMinutes}', Math.round(data.totalDurationSeconds / 60).toString())
    .replace('{averageAccuracy}', (data.averageAccuracy * 100).toFixed(1))
    .replace('{statsByGame}', formatGameStats(data.statsByGame))
    .replace('{bestDayOfWeek}', data.bestDayOfWeek);
  
  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [
      { role: 'system', content: 'You are a friendly cognitive training coach.' },
      { role: 'user', content: prompt },
    ],
    max_tokens: 500,
    temperature: 0.7,
  });
  
  const reportContent = response.choices[0].message.content;
  
  // Save to database
  await db.ai_reports.insert({
    user_id: userId,
    week_start_date: weekStart,
    week_end_date: addDays(weekStart, 6),
    total_sessions: data.totalSessions,
    total_duration_seconds: data.totalDurationSeconds,
    average_accuracy: data.averageAccuracy,
    report_content: reportContent,
    improvements: extractImprovements(reportContent),

}
```

### 6.3 Webhook Handler

```typescript
// POST /api/webhooks/stripe
export async function POST(req: Request) {
  const sig = req.headers.get('stripe-signature');
  const body = await req.text();
  
  const event = stripe.webhooks.constructEvent(body, sig, process.env.STRIPE_WEBHOOK_SECRET);
  
  switch (event.type) {
    case 'checkout.session.completed':
      const session = event.data.object;
      await updateUserSubscription(session.client_reference_id, {
        stripe_customer_id: session.customer as string,
        stripe_subscription_id: session.subscription as string,
        subscription_status: 'active',
        subscription_expires_at: new Date((session.current_period_end as number) * 1000),
      });
      break;
      
    case 'customer.subscription.deleted':
      const subscription = event.data.object;
      await updateUserSubscriptionByStripe(subscription.id, {
        subscription_status: 'canceled',
      });
      break;
  }
  
  return Response.json({ received: true });
}
```
